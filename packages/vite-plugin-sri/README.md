## Installation

```bash
pnpm i -D @keita-hino/vite-plugin-sri
```

## Usage

### add to vite config

```ts
// vite config
import { defineConfig } from 'vite';
import { sri } from '@keita-hino/vite-plugin-sri';

export default defineConfig({
	plugins: [
		sri({
			/* plugin options */
		})
	]
});
```
